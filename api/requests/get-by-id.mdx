## Overview

This endpoint retrieves complete details of a specific request by its ID. All authenticated users can access this endpoint, but employees can only view their own requests unless they have admin privileges.

## Authentication

Requires JWT authentication token in the Authorization header.

## Path Parameters

<ParamField path="id" type="integer" required>
  The unique identifier of the request
</ParamField>

## Authorization Rules

<CardGroup cols={2}>
  <Card title="Employees" icon="user">
    Can only view their own requests
  </Card>
  <Card title="Admin/Super Admin" icon="shield-halved">
    Can view any request in the system
  </Card>
</CardGroup>

## Response

Returns a detailed request object with related information.

<ResponseField name="id_solicitud" type="integer">
  Request ID
</ResponseField>

<ResponseField name="id_empleado" type="integer">
  Employee ID who created the request
</ResponseField>

<ResponseField name="id_tipo_solicitud" type="integer">
  Request type ID
</ResponseField>

<ResponseField name="id_estado_solicitud" type="integer">
  Request status ID
</ResponseField>

<ResponseField name="fecha_inicio" type="string">
  Start date (YYYY-MM-DD HH:mm:ss)
</ResponseField>

<ResponseField name="fecha_fin" type="string">
  End date (YYYY-MM-DD HH:mm:ss)
</ResponseField>

<ResponseField name="dias_solicitados" type="integer">
  Number of days requested
</ResponseField>

<ResponseField name="descripcion_solicitud" type="string">
  Request description
</ResponseField>

<ResponseField name="documento_url" type="string">
  URL to supporting document (if any)
</ResponseField>

<ResponseField name="fecha_creacion" type="string">
  Request creation timestamp
</ResponseField>

<ResponseField name="tipo_solicitud" type="object">
  Request type details
  <ResponseField name="tipo" type="string">
    Type name (e.g., "Vacaciones", "Permiso", "Ausencia")
  </ResponseField>
</ResponseField>

<ResponseField name="estado_solicitud" type="object">
  Request status details
  <ResponseField name="estado" type="string">
    Status name (e.g., "Pendiente", "Aprobada", "Rechazada")
  </ResponseField>
</ResponseField>

<ResponseField name="empleado_solicitante" type="object">
  Employee who created the request
  <ResponseField name="nombre" type="string">
    Employee name
  </ResponseField>
  <ResponseField name="correo" type="string">
    Employee email
  </ResponseField>
</ResponseField>

<ResponseField name="historial_aprobacion" type="object">
  Approval/rejection history (if processed)
  <ResponseField name="fecha_accion" type="string">
    Action timestamp
  </ResponseField>
  <ResponseField name="retroalimentacion" type="string">
    Admin feedback
  </ResponseField>
</ResponseField>

<ResponseField name="datos_perfil" type="object">
  Profile update data (only for profile update requests)
  <ResponseField name="telefono" type="string">
    Proposed phone number
  </ResponseField>
  <ResponseField name="direccion" type="string">
    Proposed address
  </ResponseField>
  <ResponseField name="img_perfil" type="string">
    Proposed profile image URL
  </ResponseField>
</ResponseField>

## Error Responses

<ResponseField name="404" type="object">
  Request not found
  ```json
  {
    "message": "Solicitud con ID 123 no encontrada."
  }
  ```
</ResponseField>

<ResponseField name="403" type="object">
  Forbidden - User doesn't have permission to view this request
  ```json
  {
    "message": "No tienes permiso para ver esta solicitud."
  }
  ```
</ResponseField>

<ResponseField name="500" type="object">
  Internal server error
  ```json
  {
    "message": "Error interno al obtener el detalle de la solicitud."
  }
  ```
</ResponseField>

## Code Examples

```bash cURL
curl -X GET 'https://api.example.com/api/solicitudes/15' \
  -H 'Authorization: Bearer YOUR_JWT_TOKEN'
```

```javascript JavaScript
const response = await fetch('https://api.example.com/api/solicitudes/15', {
  method: 'GET',
  headers: {
    'Authorization': 'Bearer YOUR_JWT_TOKEN',
    'Content-Type': 'application/json'
  }
});

const request = await response.json();
console.log(request);
```

```python Python
import requests

url = 'https://api.example.com/api/solicitudes/15'
headers = {
    'Authorization': 'Bearer YOUR_JWT_TOKEN',
    'Content-Type': 'application/json'
}

response = requests.get(url, headers=headers)
request_data = response.json()
print(request_data)
```
